# Problem

## [1345. Jump Game IV](https://leetcode.com/problems/jump-game-iv/)


# Solution 

## 1) BFS - OPTIMAL

       
      
      
   ### Code : (.cpp)
    
          class Solution {
            public:
                int minJumps(vector<int>& arr) {
                    int i, n = arr.size(), ans = 0;
                    map<int, vector<int>> mp;
                    vector<int> vis(n,0);
                    for (i=0;i<n;i++) mp[arr[i]].push_back(i);
                    queue<int> q;
                    q.push(0);
                    while (q.size()) {
                        int size = q.size();
                        while (size--) {
                            int x = q.front();
                            q.pop();
                            vis[x] = 1;
                            if (x == n-1) return ans;
                            if (x-1 >= 0 && !vis[x-1]) {
                                vis[x-1] = 1;
                                if (x-1 == n-1) return ans+1;
                                q.push(x-1);
                            }
                            if (x+1 < n && !vis[x+1]) {
                                vis[x+1] = 1;
                                if (x+1 == n-1) return ans+1;
                                q.push(x+1);
                            }
                            for (int z : mp[arr[x]]) if (!vis[z]) {
                                vis[z] = 1;
                                if (z == n-1) return ans+1;
                                q.push(z);
                            }
                            mp.erase(arr[x]);
                        }
                        ans++;
                    }
                    return ans;
                }
            };

 
      Time Complexity  : O(N) 
                         Since we traverse all the nodes once
      Space Complexity : O(N)
                         Since the maximum queue size can be max size of queue - N and a map is used.
