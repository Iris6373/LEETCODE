# Problem

## [279. Perfect Squares](https://leetcode.com/problems/perfect-squares/)


# Solution  

## 1) BFS using Queue - OPTIMAL

      
       
      
   ### Code : (.cpp)
    
          class Solution {
          public:
              int numSquares(int n) {
                  int ans = 0;
                  queue<int> q;
                  q.push(n);
                  while (q.size()) {
                      int i, j, size = q.size();
                      for (i=0;i<size;i++) {
                          int a = q.front();
                          q.pop();
                          for (j=sqrt(a);j>0;j--) {
                              if (a-j*j == 0) return ans+1;
                              else if (a-j*j > 0) q.push(a-j*j);
                          }
                      }
                      ans++;
                  }
                  return ans;
              }
          };
